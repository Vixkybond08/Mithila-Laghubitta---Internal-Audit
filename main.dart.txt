class SecurityLockPage extends StatefulWidget {
  const SecurityLockPage({Key? key}) : super(key: key);
  @override
  SecurityLockPageState createState() => SecurityLockPageState();
}

class SecurityLockPageState extends State<SecurityLockPage> {
  final TextEditingController _pinController = TextEditingController();
  int _failedAttempts = 0;
  bool _isBlocked = false;

  @override
  void initState() {
    super.initState();
    _checkBlockStatus();
  }

  Future<void> _checkBlockStatus() async {
    final prefs = await SharedPreferences.getInstance();
    String? blockedDate = prefs.getString("blocked_date");
    if (blockedDate != null) {
      String todayStr = DateTime.now().toString().split(' ')[0];
      if (blockedDate == todayStr) {
        setState(() {
          _isBlocked = true;
        });
      } else {
        await prefs.remove("blocked_date");
      }
    }
  }

  Future<void> _verifyPin() async {
    if (_isBlocked) {
      ScaffoldMessenger.of(context).showSnackBar(const SnackBar(
          content: Text("App Blocked for Today! Please try tomorrow."),
          backgroundColor: Colors.red));
      return;
    }
    DateTime now = DateTime.now();
    String correctPin = "${now.day + now.month}98";

    if (_pinController.text == correctPin) {
      Navigator.pushReplacement(context,
          MaterialPageRoute(builder: (context) => const AuditHomePage()));
    } else {
      setState(() {
        _failedAttempts++;
      });
      if (_failedAttempts >= 3) {
        final prefs = await SharedPreferences.getInstance();
        String todayStr = DateTime.now().toString().split(' ')[0];
        await prefs.setString("blocked_date", todayStr);
        setState(() {
          _isBlocked = true;
        });
        ScaffoldMessenger.of(context).showSnackBar(const SnackBar(
            content:
                Text("Maximum attempts reached! App blocked until tomorrow."),
            backgroundColor: Colors.red));
      } else {
        ScaffoldMessenger.of(context).showSnackBar(const SnackBar(
            content:
                Text("Wrong Pin.... Please contact with Mithila Laghubitta"),
            backgroundColor: Colors.red));
      }
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Column(
        children: [
          Expanded(
            child: Center(
              child: Padding(
                padding: const EdgeInsets.all(24.0),
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    const Icon(Icons.security,
                        size: 80, color: Colors.blueAccent),
                    const SizedBox(height: 15),
                    const Text("Mithila Laghubitta Audit Security",
                        style: TextStyle(fontWeight: FontWeight.bold)),
                    const SizedBox(height: 30),
                    _isBlocked
                        ? const Text(
                            "ACCESS DENIED FOR TODAY\nPlease try again tomorrow.",
                            style: TextStyle(
                                color: Colors.red, fontWeight: FontWeight.bold),
                            textAlign: TextAlign.center)
                        : TextField(
                            controller: _pinController,
                            obscureText: true,
                            keyboardType: TextInputType.number,
                            textAlign: TextAlign.center,
                            decoration: const InputDecoration(
                                hintText: "ENTER PIN",
                                border: OutlineInputBorder())),
                    const SizedBox(height: 20),
                    if (!_isBlocked)
                      ElevatedButton(
                          onPressed: _verifyPin,
                          child: const Text("Verify & Access")),
                  ],
                ),
              ),
            ),
          ),
          const Padding(
            padding: EdgeInsets.all(8.0),
            child: Text("Copyright Roshan Gurung 9845118748",
                style: TextStyle(
                    color: Colors.grey,
                    fontSize: 11,
                    fontWeight: FontWeight.bold)),
          )
        ],
      ),
    );
  }
}

class AuditHomePageState extends State<AuditHomePage> {

  String _npaStatus = "(Excellent)";
  Color _npaColor = Colors.green;
  String _prod = "0.00";
  String _spc = "0.00";

  final List<int> notes = [1000, 500, 100, 50, 20, 10, 5, 2, 1];
  Map<int, int> noteCounts = {};
  double totalCash = 0.0;

  final List<String> items = [

    "4. Staff Leave Reg/File",
    "5. Day Book",

  @override
  void initState() {
    super.initState();
    _resetData();
  }

  void _resetData() {
    for (var n in notes) noteCounts[n] = 0;
    totalCash = 0.0;
    for (int i = 0; i < items.length; i++) {
      isRaised[i] = true;
      isUpdated[i] = true;
      remarks[i] = TextEditingController();
    }
    _branchName.clear();
    _branchAddress.clear();
    _auditPFrom.clear();
    _auditPTo.clear();
    _auditDFrom.clear();
    _auditDTo.clear();
    _noOfStaffs.clear();
    _noOfCenters.clear();
    _noOfMembers.clear();
    _noOfBorrowers.clear();
    _loanPortfolio.clear();
    _savingBalance.clear();
    _overdueLoan.clear();
    _watchlist.clear();
    _npa = "0.00 %";
    _npaStatus = "(Excellent)";
    _npaColor = Colors.green;
    _prod = "0.00";
    _spc = "0.00";
  }

  void _calculate() {
    double lp = double.tryParse(_loanPortfolio.text) ?? 0;
    double ol = double.tryParse(_overdueLoan.text) ?? 0;
    double st = double.tryParse(_noOfStaffs.text) ?? 0;
    double mb = double.tryParse(_noOfMembers.text) ?? 0;
    double ct = double.tryParse(_noOfCenters.text) ?? 0;

    setState(() {
      double npaCalc = lp > 0 ? (ol / lp) * 100 : 0.0;
      _npa = "${npaCalc.toStringAsFixed(2)} %";

      if (npaCalc == 0.0) {
        _npaStatus = "(Excellent)";
        _npaColor = Colors.green;
      } else if (npaCalc > 0.0 && npaCalc < 1.0) {
        _npaStatus = "(Very Good)";
        _npaColor = Colors.green;
      } else if (npaCalc >= 1.0 && npaCalc < 2.0) {
        _npaStatus = "(Good)";
        _npaColor = Colors.green;
      } else if (npaCalc >= 2.0 && npaCalc < 3.0) {
        _npaStatus = "(Satisfactory)";
        _npaColor = Colors.green;
      } else if (npaCalc >= 3.0 && npaCalc <= 5.0) {
        _npaStatus = "(Focus on Recovery)";
        _npaColor = Colors.green;
      } else {
        _npaStatus =
            "(Daily Client Meetup, One to One Deal, Focus on Recovery)";
        _npaColor = Colors.red;
      }

      _prod = st > 0 ? (mb / st).toStringAsFixed(2) : "0.00";
      _spc = st > 0 ? (ct / st).toStringAsFixed(2) : "0.00";
    });
  }

  Future<Uint8List> _generatePdf() async {
    final pdf = pw.Document();

    List<List<String>> cashRows = [];
    notes.forEach((n) {
      if (noteCounts[n]! > 0) {
        cashRows
            .add(["Rs. $n", "${noteCounts[n]}", "Rs. ${n * noteCounts[n]!}"]);
      }
    });

    pdf.addPage(pw.MultiPage(
      build: (context) => [
        pw.Center(
            child: pw.Text("Mithila Laghubitta Bittiya Sanstha Ltd.",
                style: pw.TextStyle(
                    fontSize: 18, fontWeight: pw.FontWeight.bold))),
        pw.SizedBox(height: 12),
        pw.Text("1. Branch Details & Financial Ratios (Tabular View)",
            style: pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 12)),
        pw.SizedBox(height: 5),
        pw.TableHelper.fromTextArray(
          headers: ['Metric / Information Description', 'Value Details'],
          data: [
            ['1. Branch Name', _branchName.text],
            ['2. Branch Address', _branchAddress.text],
            ['3. Audit Period', "${_auditPFrom.text} to ${_auditPTo.text}"],
            ['4. Inspection Visit', "${_auditDFrom.text} to ${_auditDTo.text}"],
            ['5. No. of Staffs', _noOfStaffs.text],
            ['6. No. of Members', _noOfMembers.text],
            ['7. No. of Centers', _noOfCenters.text],
            ['8. No. of Borrowers', _noOfBorrowers.text],
            ['9. Loan Portfolio', "Rs. ${_loanPortfolio.text}"],
            ['10. Saving Balance', "Rs. ${_savingBalance.text}"],
            ['11. Overdue Loan', "Rs. ${_overdueLoan.text}"],
            ['12. Watchlist Portfolio', _watchlist.text],
            ['13. NPA Percentage', "$_npa $_npaStatus"],
            ['14. Staff Productivity (Member)', _prod],
            ['15. Center / Staff Ratio', _spc],
          ],
        ),
        pw.SizedBox(height: 15),
        pw.Text("2. Vault Cash Denomination Details",
            style: pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 12)),
        pw.SizedBox(height: 5),
        cashRows.isEmpty
            ? pw.Text("No physical cash inventory recorded.")
            : pw.TableHelper.fromTextArray(
                headers: ['Note Value', 'Quantity', 'Amount'], data: cashRows),
        pw.Align(
            alignment: pw.Alignment.centerRight,
            child: pw.Padding(
                padding: const pw.EdgeInsets.all(5),
                child: pw.Text("Total Cash Balance: Rs $totalCash",
                    style: pw.TextStyle(fontWeight: pw.FontWeight.bold)))),
        pw.SizedBox(height: 15),
        pw.Text("3. Audit Checklist Status",
            style: pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 12)),
        pw.SizedBox(height: 5),
        pw.TableHelper.fromTextArray(
          headers: ['Item description', 'Raised', 'Updated', 'Remarks / Logic'],
          cellStyle: const pw.TextStyle(fontSize: 9),
          headerStyle:
              pw.TextStyle(fontWeight: pw.FontWeight.bold, fontSize: 10),
          data: List.generate(items.length, (i) {
            return [
              items[i],
              isRaised[i]! ? "Yes" : "No",
              isUpdated[i]! ? "Yes" : "No",
              remarks[i]!.text
            ];
          }),
        ),
      ],
    ));
    return pdf.save();
  }

  void _clearCurrentSection() {
    setState(() {
      if (_currentStep == 0) {
        _branchName.clear();
        _branchAddress.clear();
        _auditPFrom.clear();
        _auditPTo.clear();
        _auditDFrom.clear();
        _auditDTo.clear();
        _noOfStaffs.clear();
        _noOfCenters.clear();
        _noOfMembers.clear();
        _noOfBorrowers.clear();
        _loanPortfolio.clear();
        _savingBalance.clear();
        _overdueLoan.clear();
        _watchlist.clear();
        _npa = "0.00 %";
        _npaStatus = "(Excellent)";
        _prod = "0.00";
        _spc = "0.00";
      } else if (_currentStep == 1) {
        for (var n in notes) noteCounts[n] = 0;
        totalCash = 0.0;
      } else {
        for (int i = 0; i < items.length; i++) {
          isRaised[i] = true;
          isUpdated[i] = true;
          remarks[i]!.clear();
        }
      }
    });
  }

  Future<void> _selectDate(TextEditingController c) async {
    DateTime? p = await showDatePicker(
        context: context,
        initialDate: DateTime.now(),
        firstDate: DateTime(2020),
        lastDate: DateTime(2030));
    if (p != null) setState(() => c.text = "${p.year}-${p.month}-${p.day}");
  }

  Future<void> _saveReportLocally() async {
    final data = await _generatePdf();
    final dir = await getExternalStorageDirectory();
    String bName = _branchName.text.isEmpty ? "Branch" : _branchName.text;
    String timeStamp = DateTime.now()
        .toString()
        .replaceAll(':', '-')
        .replaceAll('.', '-')
        .replaceAll(' ', '_'); // यहाँ replaceAll एरर फिक्स गरियो
    final file = File("${dir!.path}/${bName}_Audit_$timeStamp.pdf");
    await file.writeAsBytes(data);
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString("Audit_${bName}_$timeStamp", file.path);
    ScaffoldMessenger.of(context).showSnackBar(const SnackBar(
        content: Text("PDF safely saved locally in: Documents/")));
  }

class SavedReportsPage extends StatefulWidget {
  const SavedReportsPage({Key? key}) : super(key: key);
  @override
  SavedReportsPageState createState() => SavedReportsPageState();
}

class SavedReportsPageState extends State<SavedReportsPage> {
  List<String> reportKeys = [];
  Map<String, String> keyPaths = {};
  @override
  void initState() {
    super.initState();
    loadKeys();
  }

  Future<void> loadKeys() async {
    final prefs = await SharedPreferences.getInstance();
    final keys = prefs.getKeys().where((k) => k.startsWith("Audit_")).toList();
    for (var k in keys) {
      keyPaths[k] = prefs.getString(k) ?? "";
    }
    setState(() {
      reportKeys = keys;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Saved Audit PDF History")),
      body: Column(
        children: [
          Expanded(
            child: ListView.builder(
              itemCount: reportKeys.length,
              itemBuilder: (context, i) {
                return ListTile(
                  leading: const Icon(Icons.picture_as_pdf, color: Colors.red),
                  title: Text(reportKeys[i]
                      .replaceFirst("Audit_", "")
                      .replaceAll('_', ' ')),
                  onTap: () async {
                    String path = keyPaths[reportKeys[i]]!;
                    if (await File(path).exists()) {
                      await Share.shareXFiles([XFile(path)]);
                    }
                  },
                );
              },
            ),
          ),
          const Padding(
              padding: EdgeInsets.all(8.0),
              child: Text("Copyright Roshan Gurung 9845118748",
                  style: TextStyle(
                      color: Colors.grey,
                      fontSize: 11,
                      fontWeight: FontWeight.bold)))
        ],
      ),
    );
  }
}
